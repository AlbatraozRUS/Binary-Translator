mov rdx, 1
jmp pass_sort

:pass_sort
cmp rdx, 1
jne end_sort
mov rdx, 0
pop_r rax
pop_r rcx
push_r rcx
push_r rax
jmp next

:next
pop_r rax
push_r rax
cmp_r rcx, rax
je pass_sort

mov_rp rax, rcx
inc rcx
mov_rp rbx, rcx
cmp_r rax, rbx
jle next

dec rcx
mov_pr rcx, rbx
inc rcx
mov_pr rcx, rax
mov rdx, 1
jmp next

:end_sort

pop_r rcx
pop_r rax
jmp print
:print
mov_rp rbx, rax
inc rax
write rbx
cmp_r rax, rcx
jne print
